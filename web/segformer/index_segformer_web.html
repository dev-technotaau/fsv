const detectBtn = document.getElementById('detect-btn'); const recolorBtn = document.getElementById('recolor-btn'); const downloadBtn = document.getElementById('download-btn'); const resetBtn = document.getElementById('reset-btn'); // ============================================================================ // INITIALIZE // ============================================================================ async function init() { updateStatus('Loading your trained SegFormer model...', 'loading'); loader.classList.add('active'); try { // Load ONNX model session = await ort.InferenceSession.create(CONFIG.MODEL_PATH, { executionProviders: ['wasm'], graphOptimizationLevel: 'all' }); // Get model size const response = await fetch(CONFIG.MODEL_PATH); const blob = await response.blob(); const sizeMB = (blob.size / (1024 * 1024)).toFixed(2); document.getElementById('model-info').style.display = 'grid'; document.getElementById('model-status').textContent = '✓ Ready'; document.getElementById('model-input-size').textContent = `${CONFIG.INPUT_SIZE}×${CONFIG.INPUT_SIZE}`; document.getElementById('model-size').textContent = `${sizeMB} MB`; updateStatus('✓ Model loaded successfully! Upload an image to start.', 'success'); loader.classList.remove('active'); } catch (error) { console.error('Model loading error:', error); updateStatus(`✗ Failed to load model: ${error.message}`, 'error'); loader.classList.remove('active'); } } // ============================================================================ // EVENT LISTENERS // ============================================================================ fileInput.addEventListener('change', handleImageUpload); detectBtn.addEventListener('click', detectFence); recolorBtn.addEventListener('click', recolorFence); downloadBtn.addEventListener('click', downloadResult); resetBtn.addEventListener('click', reset); // Drag and drop uploadSection.addEventListener('dragover', (e) => { e.preventDefault(); uploadSection.classList.add('drag-over'); }); uploadSection.addEventListener('dragleave', () => { uploadSection.classList.remove('drag-over'); }); uploadSection.addEventListener('drop', (e) => { e.preventDefault(); uploadSection.classList.remove('drag-over'); const file = e.dataTransfer.files[0]; if (file && file.type.startsWith('image/')) { handleImageFile(file); } }); // Real-time updates opacity.addEventListener('input', () => { document.getElementById('opacity-value').textContent = opacity.value + '%'; }); threshold.addEventListener('input', () => { document.getElementById('threshold-value').textContent = threshold.value; }); colorPicker.addEventListener('input', () => { document.getElementById('color-preview').style.background = colorPicker.value; document.getElementById('color-preview').textContent = colorPicker.value; }); // ============================================================================ // IMAGE UPLOAD // ============================================================================ function handleImageUpload(event) { const file = event.target.files[0]; if (file) { handleImageFile(file); } } function handleImageFile(file) { if (file.size > 10 * 1024 * 1024) { updateStatus('✗ File too large! Max 10MB', 'error'); return; } const reader = new FileReader(); reader.onload = (e) => { const img = new Image(); img.onload = () => { originalImage = img; drawOriginalImage(); detectBtn.disabled = false; updateStatus('✓ Image loaded! Click "Detect Fence" to analyze.', 'success'); }; img.src = e.target.result; }; reader.readAsDataURL(file); } function drawOriginalImage() { originalCanvas.width = originalImage.width; originalCanvas.height = originalImage.height; const ctx = originalCanvas.getContext('2d'); ctx.drawImage(originalImage, 0, 0); } // ============================================================================ // SEGFORMER PREPROCESSING // ============================================================================ function preprocessImage(img) { const canvas = document.createElement('canvas'); canvas.width = CONFIG.INPUT_SIZE; canvas.height = CONFIG.INPUT_SIZE; const ctx = canvas.getContext('2d'); ctx.drawImage(img, 0, 0, CONFIG.INPUT_SIZE, CONFIG.INPUT_SIZE); const imageData = ctx.getImageData(0, 0, CONFIG.INPUT_SIZE, CONFIG.INPUT_SIZE); const data = imageData.data; // Normalize: (pixel / 255 - mean) / std const float32Data = new Float32Array(3 * CONFIG.INPUT_SIZE * CONFIG.INPUT_SIZE); for (let i = 0; i < CONFIG.INPUT_SIZE * CONFIG.INPUT_SIZE; i++) { const r = data[i * 4] / 255.0; const g = data[i * 4 + 1] / 255.0; const b = data[i * 4 + 2] / 255.0; float32Data[i] = (r - CONFIG.MEAN[0]) / CONFIG.STD[0]; float32Data[CONFIG.INPUT_SIZE * CONFIG.INPUT_SIZE + i] = (g - CONFIG.MEAN[1]) / CONFIG.STD[1]; float32Data[2 * CONFIG.INPUT_SIZE * CONFIG.INPUT_SIZE + i] = (b - CONFIG.MEAN[2]) / CONFIG.STD[2]; } return new ort.Tensor('float32', float32Data, [1, 3, CONFIG.INPUT_SIZE, CONFIG.INPUT_SIZE]); } // ============================================================================ // FENCE DETECTION // ============================================================================ async function detectFence() { if (!originalImage || !session) return; updateStatus('🔍 Detecting fence...', 'loading'); loader.classList.add('active'); detectBtn.disabled = true; try { const startTime = performance.now(); // Preprocess const inputTensor = preprocessImage(originalImage); if (CONFIG.DEBUG) { console.log('Input tensor shape:', inputTensor.dims); // Don't compute stats on huge arrays console.log('Input tensor sample:', inputTensor.data.slice(0, 10)); } // Run inference const feeds = { pixel_values: inputTensor }; const results = await session.run(feeds); const logits = results.logits; if (CONFIG.DEBUG) { console.log('Output logits shape:', logits.dims); console.log('Output logits sample:', logits.data.slice(0, 20)); } inferenceTime = performance.now() - startTime; document.getElementById('inference-time').textContent = `${inferenceTime.toFixed(0)}ms`; // Postprocess maskData = postprocessSegFormerLogits( logits.data, logits.dims[2], // Height logits.dims[3], // Width originalImage.width, originalImage.height ); drawMask(maskData); recolorBtn.disabled = false; updateStatus(`✓ Fence detected in ${inferenceTime.toFixed(0)}ms! Now recolor it.`, 'success'); loader.classList.remove('active'); detectBtn.disabled = false; } catch (error) { console.error('Detection error:', error); console.error('Error stack:', error.stack); updateStatus(`✗ Detection failed: ${error.message}`, 'error'); loader.classList.remove('active'); detectBtn.disabled = false; } } // ============================================================================ // POSTPROCESSING // ============================================================================ function postprocessSegFormerLogits(logits, modelHeight, modelWidth, targetWidth, targetHeight) { const numClasses = CONFIG.NUM_CLASSES; const thresholdValue = parseFloat(threshold.value); const totalPixels = modelWidth * modelHeight; if (CONFIG.DEBUG) { console.log('Postprocessing:', { modelHeight, modelWidth, targetWidth, targetHeight, numClasses, totalPixels, threshold: thresholdValue }); } // Logits shape: [1, num_classes, H, W] or [num_classes, H, W] // For SegFormer: logits are usually [1, 2, H, W] const mask = new Uint8Array(totalPixels); let positiveCount = 0; // Simple argmax approach (faster, no stack overflow) for (let i = 0; i < totalPixels; i++) { // Get logits for both classes at this pixel const bgLogit = logits[i]; // Background (class 0) const fenceLogit = logits[totalPixels + i]; // Fence (class 1) // Simple comparison: fence logit > background logit if (fenceLogit > bgLogit) { mask[i] = 1; positiveCount++; } else { mask[i] = 0; } } if (CONFIG.DEBUG) { console.log('Detection results:', { totalPixels, fencePixels: positiveCount, percentage: ((positiveCount / totalPixels) * 100).toFixed(2) + '%' }); } // Resize mask to original image size if (targetWidth !== modelWidth || targetHeight !== modelHeight) { return resizeMask(mask, modelWidth, modelHeight, targetWidth, targetHeight); } return mask; } function resizeMask(mask, srcW, srcH, dstW, dstH) { if (CONFIG.DEBUG) { console.log('Resizing mask:', { srcW, srcH, dstW, dstH }); } const resized = new Uint8Array(dstW * dstH); const xRatio = srcW / dstW; const yRatio = srcH / dstH; // Use nearest neighbor interpolation for (let y = 0; y < dstH; y++) { for (let x = 0; x < dstW; x++) { const srcX = Math.min(Math.floor(x * xRatio), srcW - 1); const srcY = Math.min(Math.floor(y * yRatio), srcH - 1); resized[y * dstW + x] = mask[srcY * srcW + srcX]; } } return resized; } function drawMask(mask) { maskCanvas.width = originalImage.width; maskCanvas.height = originalImage.height; const ctx = maskCanvas.getContext('2d'); const imageData = ctx.createImageData(originalImage.width, originalImage.height); const data = imageData.data; for (let i = 0; i < mask.length; i++) { const val = mask[i] * 255; data[i * 4] = val; // R data[i * 4 + 1] = val; // G data[i * 4 + 2] = val; // B data[i * 4 + 3] = 255; // A } ctx.putImageData(imageData, 0, 0); } // ============================================================================ // FENCE RECOLORING // ============================================================================ function recolorFence() { if (!originalImage || !maskData) return; updateStatus('🎨 Recoloring fence...', 'loading'); resultCanvas.width = originalImage.width; resultCanvas.height = originalImage.height; const ctx = resultCanvas.getContext('2d'); // Draw original ctx.drawImage(originalImage, 0, 0); // Get image data const imageData = ctx.getImageData(0, 0, originalImage.width, originalImage.height); const data = imageData.data; // Get color const rgb = hexToRgb(colorPicker.value); const alpha = parseInt(opacity.value) / 100; const mode = blendMode.value; // Apply color to masked regions for (let i = 0; i < maskData.length; i++) { if (maskData[i] === 1) { const idx = i * 4; const [r, g, b] = applyBlendMode( data[idx], data[idx + 1], data[idx + 2], rgb.r, rgb.g, rgb.b, mode ); data[idx] = r * alpha + data[idx] * (1 - alpha); data[idx + 1] = g * alpha + data[idx + 1] * (1 - alpha); data[idx + 2] = b * alpha + data[idx + 2] * (1 - alpha); } } ctx.putImageData(imageData, 0, 0); downloadBtn.disabled = false; updateStatus('✓ Recoloring complete! Download your result.', 'success'); } // ============================================================================ // BLEND MODES // ============================================================================ function applyBlendMode(r, g, b, cr, cg, cb, mode) { switch (mode) { case 'multiply': return [r * cr / 255, g * cg / 255, b * cb / 255]; case 'overlay': return [ r < 128 ? 2 * r * cr / 255 : 255 - 2 * (255 - r) * (255 - cr) / 255, g < 128 ? 2 * g * cg / 255 : 255 - 2 * (255 - g) * (255 - cg) / 255, b < 128 ? 2 * b * cb / 255 : 255 - 2 * (255 - b) * (255 - cb) / 255 ]; case 'screen': return [ 255 - (255 - r) * (255 - cr) / 255, 255 - (255 - g) * (255 - cg) / 255, 255 - (255 - b) * (255 - cb) / 255 ]; case 'hardlight': return [ cr < 128 ? 2 * r * cr / 255 : 255 - 2 * (255 - r) * (255 - cr) / 255, cg < 128 ? 2 * g * cg / 255 : 255 - 2 * (255 - g) * (255 - cg) / 255, cb < 128 ? 2 * b * cb / 255 : 255 - 2 * (255 - b) * (255 - cb) / 255 ]; default: // normal return [cr, cg, cb]; } } function hexToRgb(hex) { const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex); return result ? { r: parseInt(result[1], 16), g: parseInt(result[2], 16), b: parseInt(result[3], 16) } : { r: 139, g: 69, b: 19 }; } // ============================================================================ // DOWNLOAD & RESET // ============================================================================ function downloadResult() { const link = document.createElement('a'); link.download = 'fence_recolored.png'; link.href = resultCanvas.toDataURL(); link.click(); } function reset() { originalImage = null; maskData = null; originalCanvas.width = 0; maskCanvas.width = 0; resultCanvas.width = 0; detectBtn.disabled = true; recolorBtn.disabled = true; downloadBtn.disabled = true; fileInput.value = ''; updateStatus('Ready for new image!', 'success'); } function updateStatus(message, type = '') { status.textContent = message; status.className = 'status ' + type; } // ============================================================================ // INITIALIZE ON LOAD // ============================================================================ window.addEventListener('load', init);
